```
public String solution(int a, int b) {
        String answer[] = {"SUN","MON","TUE","WED","THU","FRI","SAT"};
        
        int total=0;
        for(int i=1;i<a;i++) {      //a에 해당하는 전월의 날짜까지의 합
        	if(i==2) {
        		total+=29;
        	}
        	else if(i==4||i==6||i==9||i==11) {
        		total+=30;
        	}else {
        		total+=31;
        	}
        }
        total+=b+4;			//2016년 1월1일은 금요일부터시작
        return answer[total%7];
    }
```
